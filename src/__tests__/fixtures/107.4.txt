module Sequencer {
    +voct pitch
    +gate gate
}
module Oscillator {
    +voct V/oct
    +cv FM1
    +cv FM2
    +audio Out
}
module LFO {
    +cv sine
}
module Envelope {
    +gate trig
    +gate gate
    +cv Out
}
module Filter {
    +audio In
    +cv freq1
    +cv freq2
    +audio LP
}
module VCA {
    +audio In
    +cv CV
    +audio Out
}

module Noise {
    +audio Out
}

module Random {
    +cv Out
}

module Mixer {
    +any In1
    +any In2
    +any In3
    +any Out
}
